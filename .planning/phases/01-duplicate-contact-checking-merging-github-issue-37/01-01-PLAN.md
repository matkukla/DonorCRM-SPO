```python
class DismissedDuplicate(TimeStampedModel):
    """Tracks dismissed duplicate pairs so they are not re-flagged."""
    contact_a = models.ForeignKey(Contact, on_delete=models.CASCADE, related_name='+')
    contact_b = models.ForeignKey(Contact, on_delete=models.CASCADE, related_name='+')
    dismissed_by = models.ForeignKey(
        settings.AUTH_USER_MODEL, on_delete=models.CASCADE,
        related_name='dismissed_duplicates'
    )

    class Meta:
        db_table = 'dismissed_duplicates'
        constraints = [
            models.UniqueConstraint(
                fields=['contact_a', 'contact_b'],
                name='unique_dismissed_pair'
            )
        ]

    def save(self, **kwargs):
        # Canonicalize: contact_a_id < contact_b_id (string comparison of UUIDs)
        if str(self.contact_a_id) > str(self.contact_b_id):
            self.contact_a_id, self.contact_b_id = self.contact_b_id, self.contact_a_id
        super().save(**kwargs)
```

1. Create `apps/contacts/services.py` with three main functions:

**find_duplicates_for_contact(contact_data, owner_id, exclude_id=None)**:
- Takes a dict with keys: first_name, last_name, email, phone
- Queries Contact.objects.filter(owner_id=owner_id, is_merged=False), excluding exclude_id if provided
- Exact email match (case-insensitive) via filter(email__iexact=email) when email is non-empty
- Exact phone match via filter(Q(phone=phone) | Q(phone_secondary=phone)) when phone is non-empty
- Fuzzy name match via TrigramSimilarity on first_name and last_name with threshold 0.4 when name is non-empty. Use Greatest() to take the higher of the two similarities.
- Score results into 3 tiers:
  - High (confidence='high'): exact email or phone match
  - Medium (confidence='medium'): name similarity >= 0.6
  - Low (confidence='low'): name similarity >= 0.4
- Return list of dicts: [{'contact': Contact, 'confidence': str, 'reasons': list[str], 'similarity': float}]
- Deduplicate by contact ID (same contact may match on multiple criteria, keep highest confidence)
- Sort: high > medium > low, then by similarity descending
- Limit to 10 results

**scan_duplicates_for_owner(owner_id)**:
- Get all non-merged contacts for owner, ordered by last_name, first_name
- Load dismissed pair set from DismissedDuplicate.objects.filter(Q(contact_a__owner_id=owner_id) | Q(contact_b__owner_id=owner_id)).values_list('contact_a_id', 'contact_b_id')
- Build canonical dismissed set: {(min(str(a), str(b)), max(str(a), str(b))) for a, b in raw_dismissed}
- For each contact, find matches among contacts with pk > current pk (avoid A-B and B-A duplication)
  - Check exact email match, exact phone match, and name similarity via TrigramSimilarity >= 0.4
- Skip pairs where canonical (min_id, max_id) is in dismissed set
- Return list of dicts: [{'contact_a': Contact, 'contact_b': Contact, 'confidence': str, 'reasons': list[str], 'similarity': float}]
- Sort by confidence descending (high, medium, low), then similarity descending

**merge_contacts(survivor_id, loser_id, field_overrides, merged_by)**:
- Decorate with `@transaction.atomic`
- Use `select_for_update()` on both contacts: survivor = Contact.objects.select_for_update().get(pk=survivor_id), same for loser
- Check loser.is_merged is False, raise ValueError('Contact has already been merged') if True
- Apply field_overrides: for each {field_name: 'right'}, copy loser's field value to survivor. Only allow these mergeable fields: first_name, last_name, email, phone, phone_secondary, organization_name, street_address, city, state, postal_code, country, status, notes
- Reassign FKs, capturing counts:
  - gifts_count = Gift.objects.filter(donor_contact=loser).update(donor_contact=survivor)
  - recurring_gifts_count = RecurringGift.objects.filter(donor_contact=loser).update(donor_contact=survivor)
  - tasks_count = Task.objects.filter(contact=loser).update(contact=survivor)
  - prayer_intentions_count = PrayerIntention.objects.filter(contact=loser).update(contact=survivor)
  - events_count = Event.objects.filter(contact=loser).update(contact=survivor)
  - journal_contacts_count = _merge_journal_contacts(survivor, loser)
- Union-merge groups: survivor.groups.add(*loser.groups.all())
- Soft-delete loser: loser.is_merged=True, loser.merged_into=survivor, loser.save(update_fields=['is_merged', 'merged_into', 'updated_at'])
- Save survivor: survivor.save() (for field overrides), then survivor.update_giving_stats()
- Create ContactMergeLog(survivor=survivor, loser_id=loser.id, loser_name=loser.full_name, merged_by=merged_by, field_overrides=field_overrides, records_migrated={'gifts': gifts_count, 'recurring_gifts': recurring_gifts_count, 'tasks': tasks_count, 'prayer_intentions': prayer_intentions_count, 'events': events_count, 'journal_contacts': journal_contacts_count})
- Clean up dismissed duplicates involving the loser: DismissedDuplicate.objects.filter(Q(contact_a=loser) | Q(contact_b=loser)).delete()
- Return survivor

**_merge_journal_contacts(survivor, loser)** (private helper):
- from apps.journals.models import JournalContact, JournalStageEvent, Decision, NextStep
- count = 0
- For each loser_jc in JournalContact.objects.filter(contact=loser):
  - survivor_jc = JournalContact.objects.filter(journal=loser_jc.journal, contact=survivor).first()
  - If survivor_jc exists (conflict):
    - JournalStageEvent.objects.filter(journal_contact=loser_jc).update(journal_contact=survivor_jc)
    - Decision.objects.filter(journal_contact=loser_jc).update(journal_contact=survivor_jc)
    - NextStep.objects.filter(journal_contact=loser_jc).update(journal_contact=survivor_jc)
    - loser_jc.delete()
  - Else (no conflict):
    - loser_jc.contact = survivor
    - loser_jc.save(update_fields=['contact'])
  - count += 1
- Return count
