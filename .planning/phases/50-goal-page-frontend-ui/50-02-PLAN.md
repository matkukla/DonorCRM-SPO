```typescript
// Props interface
interface GoalProgressBarProps {
  value: number         // 0-100+ (fill clamped to 100%, label can show >100%)
  colorVariant?: "support" | "default"  // "support" = dynamic color; "default" = primary blue
  disabled?: boolean    // muted/greyed state for empty-state rendering
  label?: string        // optional accessible label for the bar
  className?: string
}

// Color logic for Monthly Support bar:
function getSupportBarColor(pct: number): string {
  if (pct >= 100) return "bg-amber-400"    // honey gold at goal
  if (pct >= 75)  return "bg-green-500"    // green approaching goal
  return "bg-destructive"                   // red below 75%
}

// Tick mark positions: 25%, 50%, 75%, 100%
// Use: left-[25%] -translate-x-px  (shift 1px to center 2px-wide tick)
// 100% tick: left-[100%] -translate-x-full to keep inside container

// Structure sketch:
// <div className="relative w-full h-3 bg-secondary rounded-full overflow-hidden outer-track">
//   <div className="absolute h-full rounded-full transition-all duration-300 fill" style={{width: `${fillPct}%`}} />
//   {/* tick marks — on outer container, not fill */}
// </div>
// Wait — overflow-hidden on outer clips ticks too. Use separate wrapper:
// <div className="relative w-full"> {/* wrapper: no overflow-hidden */}
//   <div className="w-full h-3 bg-secondary rounded-full"> {/* track */}
//     <div className="h-full rounded-full transition-all" style={{width: `${fillPct}%`}} /> {/* fill */}
//   </div>
//   {/* tick marks as absolute divs on the wrapper */}
//   <div className="absolute top-0 h-3 w-px bg-background/70 left-[25%]" />
//   <div className="absolute top-0 h-3 w-px bg-background/70 left-[50%]" />
//   <div className="absolute top-0 h-3 w-px bg-background/70 left-[75%]" />
//   <div className="absolute top-0 h-3 w-px bg-background/70 left-[100%] -translate-x-full" />
// </div>
```

Anti-pattern: Do NOT use `overflow-hidden` on the tick-mark container — it would clip the ticks at 100%.
Anti-pattern: Do NOT position ticks on the fill div — fill width changes, ticks would shift with it.
