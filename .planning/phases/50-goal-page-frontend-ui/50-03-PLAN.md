    ```typescript
        queryKey: ["goal"],

      })

    ```

    **useUpdateGoal()** — wraps updateGoal() with cache invalidation:
    ```typescript
    export function useUpdateGoal() {
      const queryClient = useQueryClient()
      return useMutation({
        mutationFn: (payload: GoalUpdatePayload) => updateGoal(payload),
        onSuccess: (data) => {
          // Set query data directly from mutation response to avoid extra round-trip
          queryClient.setQueryData(["goal"], data)
        },
      })
    }
    ```

    Note: Using `setQueryData` instead of `invalidateQueries` avoids the stale-cache flash (Pitfall 5 from RESEARCH.md). The PATCH response already returns the updated GoalData, so we set it directly.

    Imports:
