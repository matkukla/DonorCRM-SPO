```python
class FundImportView(APIView):
    """
    POST: Import funds from CSV file (admin only)

    Query params:
        validate_only: If 'true', only validate without importing

    Expected CSV columns: fund_id, name, status

    Returns:
        - validate_only=true: {valid_count, error_count, errors}
        - import: {import_run_id, created_count, updated_count, error_count}
    """
    permission_classes = [permissions.IsAuthenticated, IsAdmin]
    parser_classes = [MultiPartParser]

    def post(self, request):
        # File validation
        if 'file' not in request.FILES:
            return Response(
                {'detail': 'No file provided.'},
                status=status.HTTP_400_BAD_REQUEST
            )

        file = request.FILES['file']
        if not file.name.endswith('.csv'):
            return Response(
                {'detail': 'File must be a CSV.'},
                status=status.HTTP_400_BAD_REQUEST
            )

        # Read and decode with BOM handling (Excel exports UTF-8 BOM)
        try:
            content = file.read().decode('utf-8-sig')
        except UnicodeDecodeError:
            return Response(
                {'detail': 'File encoding error. Please use UTF-8.'},
                status=status.HTTP_400_BAD_REQUEST
            )

        logger.info(f'Fund import started by user {request.user.email}')

        # Parse and validate
        valid_records, errors = parse_funds_csv(content, request.user)

        # Validate-only mode (dry run)
        if request.query_params.get('validate_only') == 'true':
            return Response({
                'valid_count': len(valid_records),
                'error_count': len(errors),
                'errors': errors[:20]  # Limit errors in response
            })

        # Return validation errors without importing
        if errors:
            return Response(
                {
                    'detail': 'Validation errors found. Fix errors and re-upload.',
                    'valid_count': len(valid_records),
                    'error_count': len(errors),
                    'errors': errors[:20]
                },
                status=status.HTTP_400_BAD_REQUEST
            )

        # No valid records
        if not valid_records:
            return Response(
                {'detail': 'No valid records to import.'},
                status=status.HTTP_400_BAD_REQUEST
            )

        # Create ImportRun audit record
        import_run = ImportRun.objects.create(
            type=ImportType.FUNDS,
            status=ImportStatus.IMPORTING,
            filename=file.name,
            total_rows=len(valid_records),
            uploaded_by=request.user
        )

        # Import funds
        try:
            created, updated = import_funds(valid_records, import_run)
        except Exception as e:
            import_run.status = ImportStatus.FAILED
            import_run.error_summary = {'error': str(e)}
            import_run.save()
            logger.exception(f'Fund import failed: {e}')
            return Response(
                {'detail': f'Import failed: {str(e)}'},
                status=status.HTTP_500_INTERNAL_SERVER_ERROR
            )

        logger.info(
            f'Fund import completed: {created} created, {updated} updated '
            f'(import_run {import_run.id})'
        )

        return Response({
            'import_run_id': str(import_run.id),
            'total_rows': len(valid_records),
            'created_count': created,
            'updated_count': updated,
            'error_count': 0
        })
```

```python
@pytest.mark.django_db
class TestFundImportView:
    """Integration tests for FundImportView API endpoint."""

    def test_admin_can_import_funds(self, authenticated_admin):
        """Admin can successfully import funds via POST."""
        csv_content = "fund_id,name,status\nFUND001,General Fund,active"
        csv_file = io.BytesIO(csv_content.encode('utf-8'))
        csv_file.name = 'funds.csv'

        response = authenticated_admin.post(
            '/api/v1/imports/funds/',
            {'file': csv_file},
            format='multipart'
        )

        assert response.status_code == http_status.HTTP_200_OK
        assert response.data['created_count'] == 1
        assert response.data['updated_count'] == 0
        assert 'import_run_id' in response.data

    def test_non_admin_gets_403(self, api_client, regular_user):
        """Non-admin users get 403 Forbidden."""
        api_client.force_authenticate(user=regular_user)
        csv_content = "fund_id,name,status\nFUND001,General Fund,active"
        csv_file = io.BytesIO(csv_content.encode('utf-8'))
        csv_file.name = 'funds.csv'

        response = api_client.post(
            '/api/v1/imports/funds/',
            {'file': csv_file},
            format='multipart'
        )

        assert response.status_code == http_status.HTTP_403_FORBIDDEN

    def test_missing_file_returns_400(self, authenticated_admin):
        """Missing file returns 400 with message."""
        response = authenticated_admin.post('/api/v1/imports/funds/', {}, format='multipart')

        assert response.status_code == http_status.HTTP_400_BAD_REQUEST
        assert 'No file provided' in response.data['detail']

    def test_non_csv_file_returns_400(self, authenticated_admin):
        """Non-CSV file returns 400 with message."""
        txt_file = io.BytesIO(b'not a csv')
        txt_file.name = 'funds.txt'

        response = authenticated_admin.post(
            '/api/v1/imports/funds/',
            {'file': txt_file},
            format='multipart'
        )

        assert response.status_code == http_status.HTTP_400_BAD_REQUEST
        assert 'must be a CSV' in response.data['detail']

    def test_validate_only_mode(self, authenticated_admin):
        """validate_only=true returns validation results without importing."""
        csv_content = "fund_id,name,status\nFUND001,General Fund,active\n,Missing ID,active"
        csv_file = io.BytesIO(csv_content.encode('utf-8'))
        csv_file.name = 'funds.csv'

        response = authenticated_admin.post(
            '/api/v1/imports/funds/?validate_only=true',
            {'file': csv_file},
            format='multipart'
        )

        assert response.status_code == http_status.HTTP_200_OK
        assert response.data['valid_count'] == 1
        assert response.data['error_count'] == 1
        assert Fund.objects.count() == 0  # Nothing imported

    def test_validation_errors_returns_400(self, authenticated_admin):
        """Validation errors return 400 with error details."""
        csv_content = "fund_id,name,status\n,Missing Fund ID,active"
        csv_file = io.BytesIO(csv_content.encode('utf-8'))
        csv_file.name = 'funds.csv'

        response = authenticated_admin.post(
            '/api/v1/imports/funds/',
            {'file': csv_file},
            format='multipart'
        )

        assert response.status_code == http_status.HTTP_400_BAD_REQUEST
        assert response.data['error_count'] == 1
        assert len(response.data['errors']) == 1

    def test_handles_utf8_bom(self, authenticated_admin):
        """Handles UTF-8 BOM from Excel exports."""
        # UTF-8 BOM (EF BB BF) + CSV content
        csv_content = b'\xef\xbb\xbffund_id,name,status\nFUND001,General Fund,active'
        csv_file = io.BytesIO(csv_content)
        csv_file.name = 'funds.csv'

        response = authenticated_admin.post(
            '/api/v1/imports/funds/',
            {'file': csv_file},
            format='multipart'
        )

        assert response.status_code == http_status.HTTP_200_OK
        assert response.data['created_count'] == 1
        # Verify fund_id doesn't have BOM characters
        fund = Fund.objects.get()
        assert fund.external_id == 'FUND001'

    def test_creates_import_run_audit(self, authenticated_admin):
        """Creates ImportRun record for audit trail."""
        csv_content = "fund_id,name,status\nFUND001,General Fund,active"
        csv_file = io.BytesIO(csv_content.encode('utf-8'))
        csv_file.name = 'test_funds.csv'

        response = authenticated_admin.post(
            '/api/v1/imports/funds/',
            {'file': csv_file},
            format='multipart'
        )

        assert response.status_code == http_status.HTTP_200_OK
        import_run = ImportRun.objects.get(id=response.data['import_run_id'])
        assert import_run.type == ImportType.FUNDS
        assert import_run.status == ImportStatus.COMPLETED
        assert import_run.filename == 'test_funds.csv'
        assert import_run.created_count == 1

    def test_idempotent_upsert(self, authenticated_admin):
        """Re-importing same CSV updates existing funds."""
        csv_content = "fund_id,name,status\nFUND001,General Fund,active"

        # First import
        csv_file = io.BytesIO(csv_content.encode('utf-8'))
        csv_file.name = 'funds.csv'
        response1 = authenticated_admin.post(
            '/api/v1/imports/funds/',
            {'file': csv_file},
            format='multipart'
        )
        assert response1.data['created_count'] == 1
        assert response1.data['updated_count'] == 0

        # Second import (same fund_id, different name)
        csv_content2 = "fund_id,name,status\nFUND001,Updated Fund Name,inactive"
        csv_file2 = io.BytesIO(csv_content2.encode('utf-8'))
        csv_file2.name = 'funds.csv'
        response2 = authenticated_admin.post(
            '/api/v1/imports/funds/',
            {'file': csv_file2},
            format='multipart'
        )
        assert response2.data['created_count'] == 0
        assert response2.data['updated_count'] == 1

        # Verify only one fund exists with updated data
        assert Fund.objects.count() == 1
        fund = Fund.objects.get()
        assert fund.name == 'Updated Fund Name'
        assert fund.status == 'inactive'
```
