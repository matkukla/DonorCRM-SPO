  <name>Task 2: Update ProtectedRoute to redirect with toast instead of Access Denied (QAL-08)</name>
  <files>frontend/src/components/auth/ProtectedRoute.tsx</files>
  <action>
  Replace the current "Access Denied" static div (lines 38-48 in ProtectedRoute.tsx) with a redirect to home + toast notification.

  The current code when `userLevel < requiredLevel`:
  ```tsx
  return (
    <div className="min-h-screen flex items-center justify-center">
      <div className="text-center space-y-4">
        <h1 className="text-2xl font-semibold">Access Denied</h1>
        <p className="text-muted-foreground">
          You don't have permission to view this page.
        </p>
      </div>
    </div>
  )
  ```

  Replace with a `useEffect` + `Navigate` approach to avoid toast firing on every render:

  ```tsx
  import { useEffect, useRef } from "react"
  import { Navigate, useLocation } from "react-router-dom"
  import { useAuth } from "@/providers/AuthProvider"
  import { toast } from "sonner"
  import type { ReactNode } from "react"
  ```

  In the component body, add a ref to track if toast has been shown, and a useEffect:
  ```tsx
  const toastShown = useRef(false)

  useEffect(() => {
    if (requiredRole && user) {
      const roleHierarchy: Record<string, number> = { admin: 4, finance: 3, staff: 2, read_only: 1 }
      const userLevel = roleHierarchy[user.role]
      const requiredLevel = roleHierarchy[requiredRole]
      if (userLevel < requiredLevel && !toastShown.current) {
        toastShown.current = true
        toast.info("You don't have access to that page")
      }
    }
  }, [requiredRole, user])
  ```

  And change the role check return to:
  ```tsx
  if (userLevel < requiredLevel) {
    return <Navigate to="/" replace />
  }
  ```

  The toast fires once via useRef guard, and the Navigate component handles the redirect. This prevents duplicate toasts on React re-renders while still showing the message before redirecting.

  The sidebar already hides admin nav items from non-admin users -- no sidebar changes needed (verified in research).
