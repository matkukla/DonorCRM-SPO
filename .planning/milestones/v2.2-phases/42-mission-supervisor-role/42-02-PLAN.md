```python
def get_visible_user_ids(user):
    """Return set of user IDs whose data this user can see, or None for 'all'.
    - Admin/Finance/ReadOnly: returns None (all users)
    - Mission Supervisor: returns {own_id} + {supervised user IDs}
    - Staff: returns {own_id}
    """

class IsSupervisorWriteRestricted(permissions.BasePermission):
    """Blocks writes on objects not owned by request user. Admin bypasses."""
```

```python
# Pattern A: "see-all" roles (contacts, gifts exports, prayers)
if user.role in ['admin', 'finance', 'read_only']:
    queryset = Model.objects.all()
else:
    queryset = Model.objects.filter(owner=user)

    queryset = Model.objects.all()
else:
    queryset = Model.objects.filter(owner=user)

# Pattern C: "not in see-all" (negative check)
if user.role not in ['admin', 'finance', 'read_only']:
    queryset = queryset.filter(owner=user)

    queryset = queryset.filter(owner_id=owner_id)
```

Replacement pattern for ALL:
```python
from apps.core.permissions import get_visible_user_ids

visible = get_visible_user_ids(user)
if visible is None:
    queryset = Model.objects.all()
else:
    queryset = Model.objects.filter(owner_id__in=visible)

# Owner filter (expand to supervisor):
if owner_id and user.role in ['admin', 'mission_supervisor']:
    if visible is None or int(owner_id) in visible:  # Note: UUID if needed
        queryset = queryset.filter(owner_id=owner_id)
```
