export interface ActivityHeatmapDay {
  date: string  // YYYY-MM-DD
  count: number
}

export interface ActivityHeatmapResponse {
  activities: ActivityHeatmapDay[]

export interface ActivityHeatmapParams {
  date_from?: string
  date_to?: string
}

export async function getAdminActivityHeatmap(params?: ActivityHeatmapParams): Promise<ActivityHeatmapResponse> {
  const response = await apiClient.get<ActivityHeatmapResponse>("/insights/admin/activity-heatmap/", { params })
  return response.data
}

**Create `frontend/src/pages/admin/analytics/components/TimePeriodComparison.tsx`:**

Component that compares dashboard metrics across two time periods.

- Accept `dateParams?: { date_from?: string; date_to?: string }` prop (the currently selected date range)
- Auto-calculate comparison period: if dateParams has a range, compute matching prior period:
  - If range spans ~30 days (1 month) -> compare to previous month
  - If range spans ~90 days (1 quarter) -> compare to previous quarter
  - If no dateParams (all time) -> compare current month vs last month as default
  - Calculation logic: `daysDiff = differenceInDays(to, from)`, then prior period = `{ from: subDays(from, daysDiff), to: subDays(to, daysDiff) }`
- Call `useAdminDashboardOverview(currentDateParams)` for current period
- Call `useAdminDashboardOverview(priorDateParams)` for comparison period -- use date-fns `format` to create API param strings
- Display side-by-side in a Card:
  - CardTitle: "[Current Period Label] vs [Prior Period Label]" (e.g., "Feb 1-28 vs Jan 1-28")
  - Grid of 4 MetricComparison items: Total Contacts, Conversion Rate, Stalled Contacts, Donations (12m)
- Each MetricComparison shows:
  - Label, current value (large), previous value (smaller "vs X")
  - Percentage change with trend arrow: TrendingUp (green) for positive, TrendingDown (red) for negative, Minus (gray) for no change
  - Import TrendingUp, TrendingDown, Minus from lucide-react
- Handle zero-division: if previousValue === 0 and currentValue > 0, show "+100%"; if both 0, show "0%"
- Show "Select a date range to compare periods" message if no dateParams provided
- Loading: show skeleton while either query loads
