1. Import useSearchParams from react-router-dom (add to existing react-router-dom import):

```

2. Import parseISO and isValid from date-fns (add to existing imports if any):

```

3. Inside the component, BEFORE the useState hook for dateRange, add:
```typescript
const [searchParams, setSearchParams] = useSearchParams()
```

4. Replace the existing useState initialization from:
```typescript
const [dateRange, setDateRange] = useState<DateRange | null>(null)
```

to:
```typescript
const [dateRange, setDateRange] = useState<DateRange | null>(() => {
  // Read and validate URL params on mount
  const dateFrom = searchParams.get('date_from')
  const dateTo = searchParams.get('date_to')

  if (dateFrom && dateTo) {
    const fromDate = parseISO(dateFrom)
    const toDate = parseISO(dateTo)

    // Validate both dates are valid and from is before to
    if (isValid(fromDate) && isValid(toDate) && fromDate <= toDate) {
      return { from: fromDate, to: toDate }
    }
    // Invalid dates: clear params and show console warning
    console.warn('Invalid date params in URL, ignoring:', { dateFrom, dateTo })
  }

  return null
})
```

5. Optionally (NOT required by UAT), sync URL when DateRangePicker changes by updating the setDateRange handler:
```typescript
const handleDateRangeChange = (newRange: DateRange | null) => {
  setDateRange(newRange)

  // Sync URL params
  if (newRange?.from && newRange?.to) {
    setSearchParams({
      date_from: format(newRange.from, 'yyyy-MM-dd'),
      date_to: format(newRange.to, 'yyyy-MM-dd'),
    })
  } else {
    setSearchParams({}) // Clear params
  }
}
```

Then update DateRangePicker to use the new handler:
```tsx
<DateRangePicker value={dateRange} onChange={handleDateRangeChange} />
```

WHY: Frontend currently uses component state only (line 22 in AdminAnalyticsDashboard, line 38 in StalledContacts), so invalid URL params are silently ignored because they're never read. This fix reads URL params, validates them, and initializes state accordingly. The API already handles invalid params with 400 errors, so this is purely frontend enhancement.
